üßë‚Äçüè´ HOW TO TEACH USING THIS DATA (SEQUENCE)
1 Basic SELECT
SELECT * FROM students;
SELECT name, city FROM students;
2 WHERE & Operators
SELECT * FROM students WHERE city = 'Delhi';
SELECT * FROM students WHERE age > 21;
SELECT * FROM students WHERE status = 'active' AND city = 'Mumbai';
3 ORDER BY
SELECT * FROM students ORDER BY age DESC;
4 DISTINCT
SELECT DISTINCT city FROM students;
5 LIKE
SELECT * FROM students WHERE name LIKE 'A%';
6 IS NULL
SELECT * FROM students WHERE email IS NULL;
7 Aggregate Functions
SELECT COUNT(*) FROM students;
SELECT MAX(age) FROM students;
SELECT MIN(age) FROM students;
8 GROUP BY
SELECT city, COUNT(*) FROM students GROUP BY city;
9 HAVING
SELECT city, COUNT(*) FROM students GROUP BY city HAVING COUNT(*) > 2;
10 JOINS (Real Power!)
SELECT s.name, c.course_name FROM enrollments e JOIN students s ON e.student_id = s.student_id JOIN courses c ON e.course_id = c.course_id;
