3) Operators
üìò Definition
Operators are symbols used to do calculations and comparisons.
A) Assignment Operators
üìò Definition
Used to assign or update value in a variable. 
var a = 10;
a += 5;   // a = a + 5
console.log(a);
B) Arithmetic Operators
üìò Definition
Used to do math calculations. 
var sum = 10 + 5;
var mul = 10 * 2;
var div = 10 / 2;
console.log(sum, mul, div);
C) Comparison Operators
üìò Definition
Used to compare two values and give true or false. 
console.log(5 > 3);     
console.log(5 == "5");  
console.log(5 === "5");
D) Logical Operators
üìò Definition
Used to check multiple conditions together. 
var age = 20;
var hasId = true;
console.log(age >= 18 && hasId == true);
console.log(age < 18 || hasId == true);
console.log(!hasId);

7) Array Methods
üìò Definition
Array methods help us work with array data easily.
push & pop 
var nums = [1, 2, 3];
nums.push(4);
nums.pop();
console.log(nums);
map üìò Definition
map() creates a new array by changing each value.
var nums = [1, 2, 3];
var doubled = nums.map(function(n) {
  return n * 2;
});
console.log(doubled);
filter üìò Definition
filter() keeps only values that match a condition.
var marks = [40, 60, 80];
var pass = marks.filter(function(m) {
  return m >= 60;
});
console.log(pass);
reduce üìò Definition
reduce() combines all values into one single result.
var marks = [50, 60, 70];
var total = marks.reduce(function(sum, m) {
  return sum + m;
}, 0);
console.log(total);
