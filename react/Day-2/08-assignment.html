Create a React app with 3 pages: Home, Projects, and Contact. Implement a Sidebar navigation using react-router-dom. Make sure switching pages doesn't reload the browser.

import { BrowserRouter, Routes, Route, Link } from "react-router-dom";

function App() {
  return (
    <BrowserRouter>
      <nav>
        <Link to="/">Home</Link> | 
        <Link to="/projects">Projects</Link> | 
        <Link to="/contact">Contact</Link>
      </nav>
      <Routes>
        <Route path="/" element={<h1>Home Page</h1>} />

        <Route path="/contact" element={<h1>Contact Me</h1>} />
      </Routes>
    </BrowserRouter>
  );
}

Task 3 The Toggle State

Create a "Dark Mode" toggle button. Use a state isDarkMode (boolean). When clicked, change the background color of a div from white to black and change the text from "Good Morning" to "Good Night".

üí° View Solution 
import { useState } from "react";

function ThemeToggle() {
  const [isDarkMode, setIsDarkMode] = useState(false);

  return (
    <div style={{ 
      background: isDarkMode ? "black" : "white", 
      color: isDarkMode ? "white" : "black",
      height: "200px", padding: "20px" 
    }}>
      <h1>{isDarkMode ? "Good Night üåô" : "Good Morning ‚òÄÔ∏è"}</h1>
      <button onClick={() => setIsDarkMode(!isDarkMode)}>
        Switch to {isDarkMode ? "Light" : "Dark"} Mode
      </button>
    </div>
  );
}
Task 4 The Feedback Form

Create a form with a textarea for comments and a select dropdown for 1-5 star ratings. When the form is submitted, show an alert with the user's feedback. (Don't forget preventDefault!)

üí° View Solution 
import { useState } from "react";

function FeedbackForm() {
  const [comment, setComment] = useState("");
  const [rate, setRate] = useState("5");

  const handleSubmit = (e) => {
    e.preventDefault();
    alert(`Rating: ${rate}, Comment: ${comment}`);
  };

  return (
    <form onSubmit={handleSubmit}>
      <textarea value={comment} onChange={(e) => setComment(e.target.value)} />
      <select value={rate} onChange={(e) => setRate(e.target.value)}>
        <option value="1">1 Star</option>
        <option value="5">5 Stars</option>
      </select>
      <button type="submit">Send Feedback</button>
    </form>
  );
}
